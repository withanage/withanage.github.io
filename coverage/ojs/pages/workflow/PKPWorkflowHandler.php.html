507    /**
508     * Fetch JSON-encoded editor decision options.
509     *
510     * @param array $args
511     * @param Request $request
512     *
513     * @return JSONMessage JSON object
514     */
515    public function editorDecisionActions($args, $request)
516    {
517        $this->setupTemplate($request);
518        $reviewRoundId = (int) $request->getUserVar('reviewRoundId');
519
520        // Prepare the action arguments.
521        $submission = $this->getAuthorizedContextObject(Application::ASSOC_TYPE_SUBMISSION);
522        $stageId = $this->getAuthorizedContextObject(Application::ASSOC_TYPE_WORKFLOW_STAGE);
523
524        $actionArgs = [
525            'submissionId' => $submission->getId(),
526            'stageId' => (int) $stageId,
527        ];
528
529        // If a review round was specified, include it in the args;
530        // must also check that this is the last round or decisions
531        // cannot be recorded.
532        $reviewRound = null;
533        if ($reviewRoundId) {
534            $actionArgs['reviewRoundId'] = $reviewRoundId;
535            $reviewRoundDao = DAORegistry::getDAO('ReviewRoundDAO'); /** @var ReviewRoundDAO $reviewRoundDao */
536            $lastReviewRound = $reviewRoundDao->getLastReviewRoundBySubmissionId($submission->getId(), $stageId);
537            $reviewRound = $reviewRoundDao->getById($reviewRoundId);
538        } else {
539            $lastReviewRound = null;
540        }
541
542        // If there is an editor assigned, retrieve stage decisions.
543        // Replaces StageAssignmentDAO::getEditorsAssignedToStage
544        $editorsStageAssignments = StageAssignment::withSubmissionIds([$submission->getId()])
545            ->withStageIds([$stageId])
546            ->withRoleIds([Role::ROLE_ID_MANAGER, Role::ROLE_ID_SUB_EDITOR])
547            ->get();
548
549        $user = $request->getUser();
550
551        $makeRecommendation = $makeDecision = false;
552        // if the user is assigned several times in an editorial role, check his/her assignments permissions i.e.
553        // if the user is assigned with both possibilities: to only recommend as well as make decision
554        foreach ($editorsStageAssignments as $editorsStageAssignment) {
555            if ($editorsStageAssignment->userId == $user->getId()) {
556                if (!$editorsStageAssignment->recommendOnly) {
557                    $makeDecision = true;
558                } else {
559                    $makeRecommendation = true;
560                }
561            }
562        }
563
564        // If user is not assigned to the submission,
565        // see if the user is manager, and
566        // if the group is recommendOnly
567        if (!$makeRecommendation && !$makeDecision) {
568            $userGroups = Repo::userGroup()->userUserGroups($user->getId(), $request->getContext()->getId());
569            foreach ($userGroups as $userGroup) {
570                if (in_array($userGroup->getRoleId(), [Role::ROLE_ID_MANAGER, Role::ROLE_ID_SITE_ADMIN])) {
571                    if (!$userGroup->getRecommendOnly()) {
572                        $makeDecision = true;
573                    } else {
574                        $makeRecommendation = true;
575                    }
576                }
577            }
578        }
579
580        // if the user can make recommendations, check whether there are any decisions that can be made given
581        // the stage that we are operating into.
582        $isOnlyRecommending = $makeRecommendation && !$makeDecision;
583
584        if ($isOnlyRecommending) {
585            $recommendatorsAvailableDecisions = Repo::decision()
586                ->getDecisionTypesMadeByRecommendingUsers($stageId);
587
588            if (!empty($recommendatorsAvailableDecisions)) {
589                // If there are any, then the user can be considered a decision user.
590                $makeDecision = true;
591            }
592        }
593
594        $lastRecommendation = null;
595        $allRecommendations = null;
596        $hasDecidingEditors = false;
597        if ($editorsStageAssignments->isNotEmpty() && (!$reviewRoundId || ($lastReviewRound && $reviewRoundId == $lastReviewRound->getId()))) {
598            // If this is a review stage and the user has "recommend only role"
599            if (($stageId == WORKFLOW_STAGE_ID_EXTERNAL_REVIEW || $stageId == WORKFLOW_STAGE_ID_INTERNAL_REVIEW)) {
600                if ($makeRecommendation) {
601                    // Get the made editorial decisions from the current user
602                    $editorDecisions = Repo::decision()->getCollector()
603                        ->filterBySubmissionIds([$submission->getId()])
604                        ->filterByStageIds([$stageId])
605                        ->filterByReviewRoundIds([$reviewRound->getId()])
606                        ->filterByEditorIds([$user->getId()])
607                        ->getMany();
608
609                    // Get the last recommendation
610                    foreach ($editorDecisions as $editorDecision) {
611                        if (Repo::decision()->isRecommendation($editorDecision->getData('decision'))) {
612                            if ($lastRecommendation) {
613                                if ($editorDecision->getData('dateDecided') >= $lastRecommendation->getData('dateDecided')) {
614                                    $lastRecommendation = $editorDecision;
615                                }
616                            } else {
617                                $lastRecommendation = $editorDecision;
618                            }
619                        }
620                    }
621                    if ($lastRecommendation) {
622                        $lastRecommendation = $this->getRecommendationLabel($lastRecommendation->getData('decision'));
623                    }
624
625                    // At least one deciding editor must be assigned before a recommendation can be made
626                    // Replaces StageAssignmentDAO::getDecidingEditorIds
627                    $hasDecidingEditors = StageAssignment::withSubmissionIds([$submission->getId()])
628                        ->withStageIds([$stageId])
629                        ->withRoleIds([Role::ROLE_ID_MANAGER, Role::ROLE_ID_SUB_EDITOR])
630                        ->withRecommendOnly(false)
631                        ->exists();
632
633                } elseif ($makeDecision) {
634                    // Get the made editorial decisions from all users
635                    $editorDecisions = Repo::decision()
636                        ->getCollector()
637                        ->filterBySubmissionIds([$submission->getId()])
638                        ->filterByStageIds([$stageId])
639                        ->filterByReviewRoundIds([$reviewRound->getId()])
640                        ->getMany();
641
642                    // Get all recommendations
643                    $recommendations = [];
644                    foreach ($editorDecisions as $editorDecision) {
645                        if (Repo::decision()->isRecommendation($editorDecision->getData('decision'))) {
646                            if (array_key_exists($editorDecision->getData('editorId'), $recommendations)) {
647                                if ($editorDecision->getData('dateDecided') >= $recommendations[$editorDecision->getData('editorId')]['dateDecided']) {
648                                    $recommendations[$editorDecision->getData('editorId')] = ['dateDecided' => $editorDecision->getData('dateDecided'), 'decision' => $editorDecision->getData('decision')];
649                                }
650                            } else {
651                                $recommendations[$editorDecision->getData('editorId')] = ['dateDecided' => $editorDecision->getData('dateDecided'), 'decision' => $editorDecision->getData('decision')];
652                            }
653                        }
654                    }
655                    $allRecommendations = [];
656                    foreach ($recommendations as $recommendation) {
657                        $allRecommendations[] = $this->getRecommendationLabel($recommendation['decision']);
658                    }
659                    $allRecommendations = join(__('common.commaListSeparator'), $allRecommendations);
660                }
661            }
662        }
663
664        $hasSubmissionPassedThisStage = $submission->getData('stageId') > $stageId;
665        $lastDecision = null;
666        switch ($submission->getData('status')) {
667            case PKPSubmission::STATUS_QUEUED:
668                switch ($stageId) {
669                    case WORKFLOW_STAGE_ID_SUBMISSION:
670                        if ($hasSubmissionPassedThisStage) {
671                            $lastDecision = 'editor.submission.workflowDecision.submission.underReview';
672                        }
673                        break;
674                    case WORKFLOW_STAGE_ID_INTERNAL_REVIEW:
675                    case WORKFLOW_STAGE_ID_EXTERNAL_REVIEW:
676                        if ($reviewRoundId < $lastReviewRound->getId()) {
677                            $lastDecision = 'editor.submission.workflowDecision.submission.reviewRound';
678                        } elseif ($hasSubmissionPassedThisStage) {
679                            $lastDecision = 'editor.submission.workflowDecision.submission.accepted';
680                        }
681                        break;
682                    case WORKFLOW_STAGE_ID_EDITING:
683                        if ($hasSubmissionPassedThisStage) {
684                            $lastDecision = 'editor.submission.workflowDecision.submission.production';
685                        }
686                        break;
687                }
688                break;
689            case PKPSubmission::STATUS_PUBLISHED:
690                $lastDecision = 'editor.submission.workflowDecision.submission.published';
691                break;
692            case PKPSubmission::STATUS_DECLINED:
693                $lastDecision = 'editor.submission.workflowDecision.submission.declined';
694                break;
695        }
696
697        $canRecordDecision =
698            // Only allow decisions to be recorded on the submission's current stage
699            $submission->getData('stageId') == $stageId
700
701            // Only allow decisions on the latest review round
702            && (!$lastReviewRound || $lastReviewRound->getId() == $reviewRoundId)
703
704            // At least one deciding editor must be assigned to make a recommendation
705            && ($makeDecision || $hasDecidingEditors);
706
707        $decisions = $this->getStageDecisionTypes($stageId);
708        if ($isOnlyRecommending) {
709            $decisions = Repo::decision()
710                ->getDecisionTypesMadeByRecommendingUsers($stageId);
711        }
712
713        // Assign the actions to the template.
714        $templateMgr = TemplateManager::getManager($request);
715        $templateMgr->assign([
716            'canRecordDecision' => $canRecordDecision,
717            'decisions' => $decisions,
718            'recommendations' => $this->getStageRecommendationTypes($stageId),
719            'primaryDecisions' => $this->getPrimaryDecisionTypes(),
720            'warnableDecisions' => $this->getWarnableDecisionTypes(),
721            'editorsAssigned' => $editorsStageAssignments->isNotEmpty(),
722            'stageId' => $stageId,
723            'reviewRoundId' => $reviewRound
724                ? $reviewRound->getId()
725                : null,
726            'lastDecision' => $lastDecision,
727            'lastReviewRound' => $lastReviewRound,
728            'submission' => $submission,
729            'makeRecommendation' => $makeRecommendation,
730            'makeDecision' => $makeDecision,
731            'lastRecommendation' => $lastRecommendation,
732            'allRecommendations' => $allRecommendations,
733        ]);
734        $templateMgr->registerClass('Decision', Decision::class);
735        return $templateMgr->fetchJson('workflow/editorialLinkActions.tpl');
736    }
