728    /**
729     * Filter and sanitize the request param
730     */
731    protected function _processParam(string $requestParam, mixed $value): array
732    {
733        $returnParams = [];
734        $sectionIdsQueryParam = $this->getSectionIdsQueryParam();
735        switch ($requestParam) {
736            case 'dateStart':
737            case 'dateEnd':
738            case 'timelineInterval':
739            case 'searchPhrase':
740            case 'type':
741                $returnParams[$requestParam] = $value;
742                break;
743
744            case 'count':
745                $returnParams[$requestParam] = min(100, (int) $value);
746                break;
747
748            case 'offset':
749                $returnParams[$requestParam] = (int) $value;
750                break;
751
752            case 'orderDirection':
753                $returnParams[$requestParam] = strtoupper($value);
754                break;
755
756            case $sectionIdsQueryParam:
757                if (is_string($value) && str_contains($value, ',')) {
758                    $value = explode(',', $value);
759                } elseif (!is_array($value)) {
760                    $value = [$value];
761                }
762                $returnParams['pkpSectionIds'] = array_map('intval', $value);
763                break;
764
765            case 'submissionIds':
766                if (is_string($value) && str_contains($value, ',')) {
767                    $value = explode(',', $value);
768                } elseif (!is_array($value)) {
769                    $value = [$value];
770                }
771                $returnParams[$requestParam] = array_map('intval', $value);
772                break;
773        }
774        return $returnParams;
775    }

783    protected function _processAllowedParams(array $requestParams, array $allowedParams): array

799            $returnParams['dateStart'] = $dateRange->min_date_published;
800        }
801        return $returnParams;

806     * param is also passed.

809     * request, then we only return IDs that match both conditions.

811    protected function _processSearchPhrase(string $searchPhrase, array $submissionIds = []): array
812    {
