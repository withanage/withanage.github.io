326     * Get pages converted to a nested array of page ranges
327     *
328     * For example, pages of "pp. ii-ix, 9,15-18,a2,b2-b6" will return:
329     *

340    public function getPageArray()
341    {
342        $pages = $this->getData('pages') ?? '';
343        // Strip any leading word
344        if (preg_match('/^[[:alpha:]]+\W/', $pages)) {
345            // but don't strip a leading roman numeral
346            if (!preg_match('/^[MDCLXVUI]+\W/i', $pages)) {

348                $pages = preg_replace('/^[[:alpha:]]+[:.]?/', '', $pages);

351        // strip leading and trailing space
352        $pages = trim($pages);
353        // shortcut the explode/foreach if the remainder is an empty value
354        if ($pages === '') {
