13 * @brief A class that can take a list of authorization policies, apply

130    /**

132     *
133     * @return int one of AUTHORIZATION_PERMIT or
134     *  AUTHORIZATION_DENY.
135     */
136    public function decide()
137    {
138        // Decide the root policy set which will recursively decide
139        // all nested policy sets and return a single decision.
140        $callOnDeny = null;
141        $decision = $this->_decidePolicySet($this->_rootPolicySet, $callOnDeny);
142        assert($decision !== self::AUTHORIZATION_NOT_APPLICABLE);
143
144        // Call the "call on deny" advice
145        if ($decision === AuthorizationPolicy::AUTHORIZATION_DENY && !is_null($callOnDeny)) {
146            assert(is_array($callOnDeny) && count($callOnDeny) == 3);
147            [$classOrObject, $method, $parameters] = $callOnDeny;
148            $methodCall = [$classOrObject, $method];
149            assert(is_callable($methodCall));

154    }

157    //
158    // Private helper methods

160    /**
161     * Recursively decide the given policy set.
162     *
163     * @param PolicySet $policySet
164     * @param int $callOnDeny A "call-on-deny" advice will be passed
165     *  back by reference if found.
166     *
167     * @return int one of the AUTHORIZATION_* values.
168     */
169    public function _decidePolicySet(&$policySet, &$callOnDeny)
170    {
171        // Configure the decision algorithm.
172        $combiningAlgorithm = $policySet->getCombiningAlgorithm();
173        switch ($combiningAlgorithm) {
174            case PolicySet::COMBINING_DENY_OVERRIDES:
175                $dominantEffect = AuthorizationPolicy::AUTHORIZATION_DENY;
176                $overriddenEffect = AuthorizationPolicy::AUTHORIZATION_PERMIT;
177                break;
178
179            case PolicySet::COMBINING_PERMIT_OVERRIDES:
180                $dominantEffect = AuthorizationPolicy::AUTHORIZATION_PERMIT;
181                $overriddenEffect = AuthorizationPolicy::AUTHORIZATION_DENY;
182                break;
183
184            default:
185                assert(false);
186        }
187
188        // Set the default decision.
189        $decision = $policySet->getEffectIfNoPolicyApplies();
190
191        // The following flag will record when the
192        // overridden decision state is returned by
193        // at least one policy.
194        $decidedByOverriddenEffect = false;
195
196        // Separated from below for bug #6821.
197        $context = & $this->getAuthorizedContext();
198
199        // Go through all policies within the policy set
200        // and combine them with the configured algorithm.
201        foreach ($policySet->getPolicies() as $policy) {
202            // Treat policies and policy sets differently.
203            switch (true) {
204                case $policy instanceof AuthorizationPolicy:
205                    // Make sure that the policy can access the latest authorized context.
206                    // NB: The authorized context is set by reference. This means that it
207                    // will change globally if changed by the policy which is intended
208                    // behavior so that policies can access authorized objects provided
209                    // by policies called earlier in the authorization process.
210                    $policy->setAuthorizedContext($context);
211
212                    // Check whether the policy applies.
213                    if ($policy->applies()) {
214                        // If the policy applies then retrieve its effect.
215                        $effect = $policy->effect();
216                    } else {
217                        $effect = self::AUTHORIZATION_NOT_APPLICABLE;
218                    }
219                    break;
220
221                case $policy instanceof PolicySet:
222                    // We found a nested policy set.
223                    $effect = $this->_decidePolicySet($policy, $callOnDeny);
224                    break;
225
226                default:
227                    assert(false);
228            }
229
230            // Try the next policy if this policy didn't apply.
231            if ($effect === self::AUTHORIZATION_NOT_APPLICABLE) {
232                continue;
233            }
234            assert($effect === AuthorizationPolicy::AUTHORIZATION_PERMIT || $effect === AuthorizationPolicy::AUTHORIZATION_DENY);
235
236            // "Deny" decision may cause a message to the end user.
237            if ($policy instanceof AuthorizationPolicy && $effect == AuthorizationPolicy::AUTHORIZATION_DENY
238                    && $policy->hasAdvice(AuthorizationPolicy::AUTHORIZATION_ADVICE_DENY_MESSAGE)) {
239                $this->addAuthorizationMessage($policy->getAdvice(AuthorizationPolicy::AUTHORIZATION_ADVICE_DENY_MESSAGE));
240            }
241
242            // Process the effect.
243            if ($effect === $overriddenEffect) {
244                $decidedByOverriddenEffect = true;
245            } else {
246                // In case of a "deny overrides" we allow a "call-on-deny" advice.
247                if ($policy instanceof AuthorizationPolicy && $dominantEffect == AuthorizationPolicy::AUTHORIZATION_DENY
248                        && $policy->hasAdvice(AuthorizationPolicy::AUTHORIZATION_ADVICE_CALL_ON_DENY)) {
249                    $callOnDeny = $policy->getAdvice(AuthorizationPolicy::AUTHORIZATION_ADVICE_CALL_ON_DENY);
250                }
251
252                // Only one dominant effect overrides all other effects
253                // so we don't even have to evaluate other policies.
254                return $dominantEffect;
255            }
256        }
257
258        // Only return an overridden effect if at least one
259        // policy returned that effect and none returned the
260        // dominant effect.
261        if ($decidedByOverriddenEffect) {
262            $decision = $overriddenEffect;
263        }
264        return $decision;
265    }
266}
