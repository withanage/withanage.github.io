336     * Parse children nodes in the install descriptor.
337     *
338     * @param XMLNode $installTree
339     */
340    public function parseInstallNodes($installTree)
341    {
342        foreach ($installTree->getChildren() as $node) {
343            switch ($node->getName()) {
344                case 'schema':
345                case 'data':
346                case 'code':
347                case 'migration':
348                case 'note':
349                    $this->addInstallAction($node);
350                    break;
351                case 'upgrade':
352                    $minVersion = $node->getAttribute('minversion');
353                    $maxVersion = $node->getAttribute('maxversion');
354                    if ((!isset($minVersion) || $this->currentVersion->compare($minVersion) >= 0) && (!isset($maxVersion) || $this->currentVersion->compare($maxVersion) <= 0)) {
355                        $this->parseInstallNodes($node);
356                    }
357                    break;
358            }
359        }
360    }

362    /**
363     * Add an installer action from the descriptor.
364     *
365     * @param XMLNode $node
366     */
367    public function addInstallAction($node)
368    {
369        $fileName = $node->getAttribute('file');
370
371        if (!isset($fileName)) {
372            $this->actions[] = ['type' => $node->getName(), 'file' => null, 'attr' => $node->getAttributes()];
373        } elseif (strstr($fileName, '{$installedLocale}')) {
374            // Filename substitution for locales
375            foreach ($this->installedLocales as $thisLocale) {
376                $newFileName = str_replace('{$installedLocale}', $thisLocale, $fileName);
377                $this->actions[] = ['type' => $node->getName(), 'file' => $newFileName, 'attr' => $node->getAttributes()];
378            }
379        } else {
380            $newFileName = str_replace('{$locale}', $this->locale, $fileName);
381            if (!file_exists($newFileName)) {
382                // Use version from default locale if data file is not available in the selected locale
383                $newFileName = str_replace('{$locale}', self::INSTALLER_DEFAULT_LOCALE, $fileName);
384            }
385
386            $this->actions[] = ['type' => $node->getName(), 'file' => $newFileName, 'attr' => $node->getAttributes()];
387        }
388    }
