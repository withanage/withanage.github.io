74    /**
75     * Remove Unique Clicks
76     * If multiple transactions represent the same item and occur in the same user-sessions, only one unique activity MUST be counted for that item.
77     * Unique item is a submission.
78     * A user session is defined by the combination of IP address + user agent + transaction date + hour of day.
79     * Only the last unique activity will be retained (and thus counted), all the other will be removed.
80     *
81     * See https://www.projectcounter.org/code-of-practice-five-sections/7-processing-rules-underlying-counter-reporting-data/#counting
82     */

87                "
88                DELETE FROM {$this->table} usui
89                WHERE EXISTS (
90                    SELECT * FROM (
91                        SELECT 1 FROM {$this->table} usuit
92                        WHERE usui.load_id = ? AND usuit.load_id = usui.load_id AND
93                        usuit.context_id = usui.context_id AND
94                        usuit.ip = usui.ip AND
95                        usuit.user_agent = usui.user_agent AND
96                        usuit.submission_id = usui.submission_id AND
