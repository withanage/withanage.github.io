81    /**
82     * Remove Double Clicks according to COUNTER guidelines
83     * Remove the potential of over-counting which could occur when a user clicks the same link multiple times.
84     * Double-clicks, i.e. two clicks in succession, on a link by the same user within a 30-second period MUST be counted as one action.
85     * When two actions are made for the same URL within 30 seconds the first request MUST be removed and the second retained.
86     * A user is identified by IP address combined with the browsers user-agent.
87     *
88     * See https://www.projectcounter.org/code-of-practice-five-sections/7-processing-rules-underlying-counter-reporting-data/#doubleclick
89     */
90    public function removeDoubleClicks(string $loadId, int $counterDoubleClickTimeFilter): void
91    {
92        if (substr(Config::getVar('database', 'driver'), 0, strlen('postgres')) === 'postgres') {
93            DB::statement(
94                "
95                DELETE FROM {$this->table} ust
96                WHERE EXISTS (
97                    SELECT * FROM (
98                        SELECT 1 FROM {$this->table} ustt
99                        WHERE ust.load_id = ? AND ustt.load_id = ust.load_id AND
100                        ustt.context_id = ust.context_id AND
101                        ustt.ip = ust.ip AND ustt.user_agent = ust.user_agent AND ustt.canonical_url = ust.canonical_url AND
102                        EXTRACT(EPOCH FROM (ustt.date - ust.date)) < ? AND
103                        EXTRACT(EPOCH FROM (ustt.date - ust.date)) > 0 AND
104                        ust.line_number < ustt.line_number) AS tmp
105                )
106                ",
107                [$loadId, $counterDoubleClickTimeFilter]
108            );
109        } else {
110            DB::statement(
111                "
112                DELETE FROM ust USING {$this->table} ust
113                INNER JOIN {$this->table} ustt ON (
114                    ustt.load_id = ust.load_id AND
115                    ustt.context_id = ust.context_id AND
116                    ustt.ip = ust.ip AND
117                    ustt.user_agent = ust.user_agent AND
118                    ustt.canonical_url = ust.canonical_url
119                )
120                WHERE ust.load_id = ? AND
121                    TIMESTAMPDIFF(SECOND, ust.date, ustt.date) < ? AND
122                    TIMESTAMPDIFF(SECOND, ust.date, ustt.date) > 0 AND
123                    ust.line_number < ustt.line_number
124                ",
125                [$loadId, $counterDoubleClickTimeFilter]
126            );
127        }
128    }
